import React from "react";
import { useTheme } from "../themes/ThemeProvider";

export interface IconButtonProps {
  label: string;
  active: boolean;
  onClick?: () => void;
}

export const IconButton: React.FC<IconButtonProps> = ({
  label,
  active,
  onClick,
}) => {
  const { theme } = useTheme();

  const style: React.CSSProperties = active
    ? {
        borderColor: theme.iconButton.activeBorder,
        background: theme.iconButton.activeColor,
        color: theme.iconButton.hoverText,
        boxShadow: theme.iconButton.activeShadow,
        transform: "scale(1.02)",
      }
    : {
        borderColor: theme.iconButton.activeBorder,
        background: theme.iconButton.inactiveColor,
        color: theme.iconButton.hoverText,
        boxShadow: "none",
        transform: "scale(1.0)",
      };

  return (
    <button
      type="button"
      onClick={onClick}
      className="
        w-full
        py-2 px-4
        rounded
        border
        flex items-center justify-center
        text-sm font-bold
        cursor-pointer select-none
        transition-all duration-100 ease-out
        hover:scale-105
        focus:outline-none focus:ring-1
      "
      style={{
        ...style,
        fontFamily: theme.core.bodyFont,
      }}
    >
      {label}
    </button>
  );
};
